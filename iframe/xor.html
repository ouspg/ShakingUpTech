function binaariksi(data) {
  if (typeof data === "string") {
    let binaari = "";
    for (var i = 0; i < data.length; i++) {
      binaari += "0" + data[i].charCodeAt(0).toString(2) + " ";
    }
    return binaari.trim();
  } else if (typeof data === "number") {
    return (data >>> 0).toString(2);
  }
}
